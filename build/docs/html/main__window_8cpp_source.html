113 void MainWindow::loadStations() {
114 statusLabel->setText("Ładowanie stacji...");
115
116 // Sprawdzenie dostępności API
117 if (!apiClient->isApiAvailable()) {
118 statusLabel->setText("API niedostępne, próba wczytania lokalnych danych");
119
120 // Sprawdzenie czy plik z danymi istnieje
121 QFileInfo checkFile("data/stations.json");
122 if (checkFile.exists() && checkFile.isFile()) {
123 stations = apiClient->loadStationsFromFile();
124
125 if (stations.empty()) {
126 QMessageBox::warning(this, "Błąd", "Nie udało się wczytać danych z pliku lokalnego");
127 statusLabel->setText("Błąd wczytywania danych");
128 return;
129 }
130 } else {
131 QMessageBox::warning(this, "Błąd", "API niedostępne, a brak lokalnego pliku z danymi");
132 statusLabel->setText("Brak danych");
133 return;
134 }
135 } else {
136 // Pobranie stacji z API
137 stations = apiClient->getAllStations();
138
139 if (stations.empty()) {
140 QMessageBox::warning(this, "Błąd", "Nie udało się pobrać stacji pomiarowych");
141 statusLabel->setText("Błąd pobierania danych");
142 return;
143 }
144
145 // Upewnienie się że katalog istnieje
146 QDir().mkpath("data");
147
148 // Zapisanie stacji do pliku
149 apiClient->saveStationsToFile();
150 }
151
152 // Wypełnienie ComboBox'a stacjami
153 stationComboBox->clear();
154 for (const auto& station : stations) {
155 QString displayText = QString("%1 (%2, %3)").arg(
156 QString::fromStdString(station.name),
157 QString::fromStdString(station.city),
158 QString::fromStdString(station.province)
159 );
160 stationComboBox->addItem(displayText);
161 }
162
163 statusLabel->setText("Gotowy");
164 }

237 // Czyszczenie tabeli

240 // Ustawienie liczby wierszy
241 dataTable->setRowCount(measurements.size());
242
243 // Wypełnienie tabeli danymi
244 for (size_t i = 0; i < measurements.size(); ++i) {
245 const auto& measurement = measurements[i];
246
247 // Data i czas
248 QDateTime dateTime = QDateTime::fromString(QString::fromStdString(measurement.date), Qt::ISODate);
249 QTableWidgetItem *dateItem = new QTableWidgetItem(dateTime.toString("dd.MM.yyyy hh:mm"));
250 dataTable->setItem(i, 0, dateItem);
251
252 // Wartość
253 QTableWidgetItem *valueItem = new QTableWidgetItem(QString::number(measurement.value, 'f', 2));
254 dataTable->setItem(i, 1, valueItem);
255 }
256
257 // Sortowanie po dacie (malejąco)
258 dataTable->sortItems(0, Qt::DescendingOrder);

261 void MainWindow::createChart() {
262 if (measurements.empty()) {
263 QMessageBox::information(this, "Informacja", "Brak danych do wyświetlenia na wykresie");
264 return;
265 }
266
267 // Pobranie nazwy parametru
268 int sensorIndex = sensorComboBox->currentIndex();
269 if (sensorIndex < 0 || sensorIndex >= static_cast<int>(sensors.size())) {
270 QMessageBox::warning(this, "Błąd", "Nie wybrano parametru");
271 return;
272 }
273
274 // Sprawdź czy chartView jest poprawnie zainicjalizowany
275 if (!chartView) {
276 QMessageBox::critical(this, "Błąd", "Błąd komponentu wykresu");
277 return;
278 }
279
280 QString paramName = QString::fromStdString(sensors[sensorIndex].paramName);
281 QString paramFormula = QString::fromStdString(sensors[sensorIndex].paramFormula);
282
283 // Utwórz wykres
284 QChart *chart = new QChart();
285
286 try {
287 // Debug info
288 std::cout << "Tworzenie wykresu dla parametru: " << sensors[sensorIndex].paramName
289 << " (" << sensors[sensorIndex].paramFormula << ")" << std::endl;
290 std::cout << "Liczba pomiarów: " << measurements.size() << std::endl;
291
292 // Utworzenie serii danych
293 QLineSeries *series = new QLineSeries();
294 series->setName(paramName);
295
296 // Wartości min i max do ustalenia zakresu osi Y
297 double minValue = std::numeric_limits<double>::max();
298 double maxValue = std::numeric_limits<double>::lowest();
299 qint64 minTime = std::numeric_limits<qint64>::max();
300 qint64 maxTime = std::numeric_limits<qint64>::lowest();
301
302 // Wypełnienie serii danymi
303 for (const auto& measurement : measurements) {
304 // Konwersja daty na timestamp
305 QDateTime dateTime = QDateTime::fromString(QString::fromStdString(measurement.date), Qt::ISODate);
306 qint64 timestamp = dateTime.toMSecsSinceEpoch();
307
308 // Dodanie punktu do serii
309 series->append(timestamp, measurement.value);
310
311 // Debug info dla pierwszych 5 punktów
312 if (series->count() <= 5) {
313 std::cout << "Punkt " << series->count() << ": "
314 << dateTime.toString("yyyy-MM-dd hh:mm").toStdString()
315 << ", wartość: " << measurement.value << std::endl;
316 }
317
318 // Aktualizacja wartości min/max
319 minValue = std::min(minValue, measurement.value);
320 maxValue = std::max(maxValue, measurement.value);
321 minTime = std::min(minTime, timestamp);
322 maxTime = std::max(maxTime, timestamp);
323 }
324
325 if (series->count() == 0) {
326 QMessageBox::warning(this, "Błąd", "Nie udało się utworzyć punktów wykresu");
327 delete series;
328 delete chart;
329 return;
330 }
331
332 // Dodanie marginesu do zakresu wartości
333 double margin = (maxValue - minValue) * 0.1;
334 if (margin < 0.001) margin = 0.1; // Minimalny margines
335 minValue = minValue - margin;
336 maxValue = maxValue + margin;
337
338 // Określenie zakresu dat dla tytułu
339 QDateTime minDateTime = QDateTime::fromMSecsSinceEpoch(minTime);
340 QDateTime maxDateTime = QDateTime::fromMSecsSinceEpoch(maxTime);
341 QString dateRangeStr = QString("Okres: %1 - %2").arg(
342 minDateTime.toString("dd.MM.yyyy"),
343 maxDateTime.toString("dd.MM.yyyy")
344 );
345
346 // Dodanie serii do wykresu
347 chart->addSeries(series);
348 // Połączenie tytułu z zakresem dat
349 chart->setTitle(QString("Pomiary: %1 (%2)\n%3").arg(
350 paramName,
351 paramFormula,
352 dateRangeStr
353 ));
354
355 // Osie wykresu
356 QDateTimeAxis *axisX = new QDateTimeAxis;
357 axisX->setTickCount(8); // Mniej punktów na osi
358 axisX->setFormat("hh:mm"); // Tylko godzina i minuty
359 axisX->setTitleText("Czas pomiaru");
360
361 QValueAxis *axisY = new QValueAxis;
362 axisY->setLabelFormat("%.2f");
363 axisY->setTitleText(paramFormula);
364 axisY->setRange(minValue, maxValue);
365
366 chart->addAxis(axisX, Qt::AlignBottom);
367 chart->addAxis(axisY, Qt::AlignLeft);
368 series->attachAxis(axisX);
369 series->attachAxis(axisY);
370
371 // Ustawienie zakresu czasu
372 axisX->setRange(QDateTime::fromMSecsSinceEpoch(minTime), QDateTime::fromMSecsSinceEpoch(maxTime));
373
374 // Ustawienie nowego wykresu w widoku
375 chartView->setChart(chart);
376
377 // Animacja
378 chart->setAnimationOptions(QChart::SeriesAnimations);
379
380 std::cout << "Wykres utworzony pomyślnie" << std::endl;
381 }
382 catch (const std::exception& e) {
383 QMessageBox::critical(this, "Błąd", QString("Wystąpił błąd podczas tworzenia wykresu: %1").arg(e.what()));
384 delete chart;
385 }
386 }
